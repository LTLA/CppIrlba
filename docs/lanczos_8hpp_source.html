39/*
40 * W is a matrix with number of rows equal to `mat.rows()`.
41 * The size of the working subspace is defined as the number of columns.
42 * The first `start` columns should contain orthonormal column vectors with non-zero L2 norms.
43 * On output, the rest of `W` is filled with orthonormal vectors.
44 *
45 * V is a matrix with number of rows equal to `mat.cols()` and number of columns equal to `W.cols()`.
46 * The first `start + 1` columns should contain orthonormal column vectors with non-zero L2 norms.
47 * On output, the rest of `V` is filled with orthonormal vectors.
48 *
49 * B is a square matrix with number of rows and columns equal to the size of the working subspace.
50 * On output, B is filled with upper diagonal entries, starting from the `start`-th row/column.
51 */
52template<class Matrix_, class EigenMatrix_, class EigenVector_, class Engine_>
53void run_lanczos_bidiagonalization(
54 const Matrix_& mat,
55 EigenMatrix_& W,
56 EigenMatrix_& V,
57 EigenMatrix_& B,
58 Engine_& eng,
59 LanczosWorkspace<EigenVector_, Matrix_>& inter,
60 Eigen::Index start,
61 const Options& options)
62{
63 typedef typename EigenMatrix_::Scalar Float;
64 Float raw_eps = options.invariant_subspace_tolerance;
65 Float eps = (raw_eps < 0 ? std::pow(std::numeric_limits<Float>::epsilon(), 0.8) : raw_eps);
66
67 Eigen::Index work = W.cols();
68 auto& F = inter.F;
69 auto& W_next = inter.W_next;
70 auto& otmp = inter.orthog_tmp;
71
72 F = V.col(start);
73 wrapped_multiply(mat, F, inter.work, W_next); // i.e., W_next = mat * F;
74
75 // If start = 0, there's nothing to orthogonalize against.
76 if (start) {
77 orthogonalize_vector(W, W_next, start, otmp);
78 }
79
80 Float S = W_next.norm();
81 if (S < eps) {
82 throw std::runtime_error("starting vector near the null space of the input matrix");
83 }
84 W_next /= S;
85 W.col(start) = W_next;
86
87 // The Lanczos iterations themselves, see algorithm 2.1 of Baglama and Reichel.
88 for (Eigen::Index j = start; j < work; ++j) {
89 wrapped_adjoint_multiply(mat, W.col(j), inter.awork, F); // i.e., F = mat.adjoint() * W.col(j);
90
91 F -= S * V.col(j); // equivalent to daxpy.
92 orthogonalize_vector(V, F, j + 1, otmp);
93
94 if (j + 1 < work) {
95 Float R_F = F.norm();
96
97 if (R_F < eps) {
98 fill_with_random_normals(F, eng);
99 orthogonalize_vector(V, F, j + 1, otmp);
100 R_F = F.norm();
101 F /= R_F;
102 R_F = 0;
103 } else {
104 F /= R_F;
105 }
106
107 V.col(j + 1) = F;
108 B(j, j) = S;
109 B(j, j + 1) = R_F;
110
111 wrapped_multiply(mat, F, inter.work, W_next); // i.e., W_next = mat * F;
112 W_next -= R_F * W.col(j); // equivalent to daxpy.
113
114 // Full re-orthogonalization, using the left-most 'j + 1' columns of W.
115 // Recall that W_next will be the 'j + 2'-th column, i.e., W.col(j + 1) in
116 // 0-indexed terms, so we want to orthogonalize to all previous columns.
117 orthogonalize_vector(W, W_next, j + 1, otmp);
118
119 S = W_next.norm();
120 if (S < eps) {
121 fill_with_random_normals(W_next, eng);
122 orthogonalize_vector(W, W_next, j + 1, otmp);
123 S = W_next.norm();
124 W_next /= S;
125 S = 0;
126 } else {
127 W_next /= S;
128 }
129
130 W.col(j + 1) = W_next;
131 } else {
132 // The paper's algorithm sets B(j + 1, j + 1) = S in the
133 // j + 1 < work clause. But the irlba R package's C code
134 // sets it on the last loop, which gives the same result;
135 // so to avoid any headaches, we do the same thing too.
136 B(j, j) = S;
137 }
138 }
139}
