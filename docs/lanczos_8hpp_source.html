161  constexpr bool do_center = !std::is_same<CENTER, bool>::value;
162  constexpr bool do_scale = !std::is_same<SCALE, bool>::value;
163 
164  // Applying the scaling.
165  if constexpr(do_scale) {
166  F = F.cwiseQuotient(scale);
167  }
168

170

182  if (ncols) {
