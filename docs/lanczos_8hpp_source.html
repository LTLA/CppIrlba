151 W.col(start) = W_next;

153 // The Lanczos iterations themselves.

156
157 F -= S * V.col(j); // equivalent to daxpy.
158 orthogonalize_vector(V, F, j + 1, otmp);
159
160 if (j + 1 < work) {
161 double R_F = F.norm();
162

165 orthogonalize_vector(V, F, j + 1, otmp);
166 R_F = F.norm();

168 R_F = 0;
169 } else {
170 F /= R_F;

179
