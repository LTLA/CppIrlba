82 constexpr bool do_scale = !std::is_same<SCALE, bool>::value;
83
84 int work = W.cols();
85 orthog.set_size(work);
86 F.resize(mat.cols());
87 W_next.resize(mat.rows());
88

93 double S = W_next.norm();

98 W.col(start) = W_next;

100 // The Lanczos iterations themselves.
101 for (int j = start; j < work; ++j) {
102 F.noalias() = mat.adjoint() * W.col(j);
103
104 // Centering and scaling, if requested.
105 if constexpr(do_center) {

111 }

114 orthog.run(V, F, j + 1);

121 orthog.run(V, F, j + 1);
122 R_F = F.norm();
123 F /= R_F;
124 R_F = 0;
125 } else {
126 F /= R_F;
127 }

132 B(j, j + 1) = R_F;

143 } else {

148 } else {

157 /* This function just updates W_next based on the current F, i.e., the
158 * initialization vector (when ncols = 0) or the residual vector
159 * (otherwise). It will also mutate F if centering or scaling is requested.
160 *
161 * It's worth noting that, when ncols > 0, 'F' at this point is equivalent
162 * to 'x', the temporary buffer used in irlb.c's inner loop. 'F's original
163 * value will not be used in the rest of the loop, so it's safe to mutate
164 * 'F' inside the function here.
165 */
166 template<class M, class CENTER, class SCALE>
167 void update_W_next(const M& mat, const CENTER& center, const SCALE& scale, const Eigen::MatrixXd& W, int ncols) {
168 constexpr bool do_center = !std::is_same<CENTER, bool>::value;
169 constexpr bool do_scale = !std::is_same<SCALE, bool>::value;
170
171 // Applying the scaling.
172 if constexpr(do_scale) {
173 F = F.cwiseQuotient(scale);
174 }
175
176 W_next.noalias() = mat * F;
177
178 // Applying the centering.
179 if constexpr(do_center) {
180 double beta = F.dot(center);
181 for (auto& w : W_next) {
182 w -= beta;
183 }
184 }
185
186 // Full re-orthogonalization, using the left-most 'ncol' columns of W.
187 // Recall that W_next will be the 'ncol + 1' column, i.e., W.col(ncol) in
188 // 0-indexed terms, so we want to orthogonalize to all previous columns.
189 if (ncols) {
190 orthog.run(W, W_next, ncols);
191 }
192 return;
193 }
