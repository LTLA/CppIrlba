71 const Eigen::Index nr = matrix.rows();
72 const Eigen::Index nc = matrix.cols();

74 EigenVector_ center0;
75 if (center) {
76 if (nr < 1) {
77 throw std::runtime_error("cannot center with no observations");
78 }
79 center0.resize(nc);
80 }
81
82 EigenVector_ scale0;
83 if (scale) {
84 if (nr < 2) {
85 throw std::runtime_error("cannot scale with fewer than two observations");

87 scale0.resize(nc);
88 }

90 for (Eigen::Index i = 0; i < nc; ++i) {
91 typename EigenVector_::Scalar mean = 0;
92 if (center) {
93 mean = matrix.col(i).sum() / nr;
94 center0[i] = mean;
95 }
96 if (scale) {
97 EigenVector_ current = matrix.col(i); // force it to be a Vector, even if it's a sparse matrix.
98 typename EigenVector_::Scalar var = 0;
99 for (auto x : current) {
100 var += (x - mean)*(x - mean);
101 }
102
103 if (var) {
104 scale0[i] = std::sqrt(var/(nr - 1));
105 } else {
106 scale0[i] = 1;
107 }
108 }
109 }

115 return compute<Interface>(centered_scaled, number, outU, outV, outD, options);
116 } else {

118 }
119 } else {
