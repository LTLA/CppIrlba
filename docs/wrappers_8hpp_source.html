317 void multiply(const Right& rhs, Workspace& work, Eigen::VectorXd& out) const {
318 // We store the result here, because the underlying matrix's multiply()
319 // might need to access rhs/scale multiple times, especially if it's
320 // parallelized. Better to pay the cost of accessing a separate memory
321 // space than computing the quotient repeatedly.
