183 void fragment_threads() {
184 auto total_nzeros = ptrs[primary_dim]; // last element - not using back() to avoid an extra requirement on PointerArray.
185 PointerType per_thread = std::ceil(static_cast<double>(total_nzeros) / nthreads);
186
187 // Splitting columns across threads so each thread processes the same number of nonzero elements.
188 primary_starts.resize(nthreads);
189 primary_ends.resize(nthreads);
190 {
191 size_t primary_counter = 0;
192 PointerType sofar = per_thread;
193 for (int t = 0; t < nthreads; ++t) {
194 primary_starts[t] = primary_counter;
195 while (primary_counter < primary_dim && ptrs[primary_counter + 1] <= sofar) {
196 ++primary_counter;
197 }
198 primary_ends[t] = primary_counter;
199 sofar += per_thread;
200 }
201 }
202
203 // Splitting rows across threads so each thread processes the same number of nonzero elements.
204 secondary_nonzero_starts.resize(nthreads + 1, std::vector<PointerType>(primary_dim));
205 {
206 std::vector<PointerType> secondary_nonzeros(secondary_dim);
207 for (PointerType i = 0; i < total_nzeros; ++i) { // don't using range for loop to avoid an extra requirement on IndexArray.
208 ++(secondary_nonzeros[indices[i]]);
209 }
210
211 std::vector<IndexType> secondary_ends(nthreads);
212 IndexType secondary_counter = 0;
213 PointerType sofar = per_thread;
214 PointerType cum_rows = 0;
215
216 for (int t = 0; t < nthreads; ++t) {
217 while (secondary_counter < secondary_dim && cum_rows <= sofar) {
218 cum_rows += secondary_nonzeros[secondary_counter];
219 ++secondary_counter;
220 }
221 secondary_ends[t] = secondary_counter;
222 sofar += per_thread;
223 }
224
225 for (size_t c = 0; c < primary_dim; ++c) {
226 auto primary_start = ptrs[c], primary_end = ptrs[c + 1];
227 secondary_nonzero_starts[0][c] = primary_start;
228
229 auto s = primary_start;
230 for (int thread = 0; thread < nthreads; ++thread) {
231 while (s < primary_end && indices[s] < secondary_ends[thread]) {
232 ++s;
233 }
234 secondary_nonzero_starts[thread + 1][c] = s;
235 }
236 }
237 }
