185 void fragment_threads() {
186 auto total_nzeros = my_ptrs[my_primary_dim]; // last element - not using back() to avoid an extra requirement on PointerArray.
187 PointerType per_thread = (total_nzeros / my_nthreads) + (total_nzeros % my_nthreads > 0); // i.e., ceiling.
188
189 // Splitting columns across threads so each thread processes the same number of nonzero elements.
190 my_primary_starts.resize(my_nthreads);
191 my_primary_ends.resize(my_nthreads);
192 {
193 size_t primary_counter = 0;
194 PointerType sofar = per_thread;
195 for (int t = 0; t < my_nthreads; ++t) {
196 my_primary_starts[t] = primary_counter;
197 while (primary_counter < my_primary_dim && my_ptrs[primary_counter + 1] <= sofar) {
198 ++primary_counter;
199 }
200 my_primary_ends[t] = primary_counter;
201 sofar += per_thread;
202 }
203 }
204
205 // Splitting rows across threads so each thread processes the same number of nonzero elements.
206 my_secondary_nonzero_starts.resize(my_nthreads + 1, std::vector<PointerType>(my_primary_dim));
207 {
208 std::vector<PointerType> secondary_nonzeros(my_secondary_dim);
209 for (PointerType i = 0; i < total_nzeros; ++i) { // don't using range for loop to avoid an extra requirement on IndexArray.
210 ++(secondary_nonzeros[my_indices[i]]);
211 }
212
213 std::vector<IndexType> secondary_ends(my_nthreads);
214 IndexType secondary_counter = 0;
215 PointerType sofar = per_thread;
216 PointerType cum_rows = 0;
217
218 for (int t = 0; t < my_nthreads; ++t) {
219 while (secondary_counter < my_secondary_dim && cum_rows <= sofar) {
220 cum_rows += secondary_nonzeros[secondary_counter];
221 ++secondary_counter;
222 }
223 secondary_ends[t] = secondary_counter;
224 sofar += per_thread;
225 }
226
227 for (size_t c = 0; c < my_primary_dim; ++c) {
228 auto primary_start = my_ptrs[c], primary_end = my_ptrs[c + 1];
229 my_secondary_nonzero_starts[0][c] = primary_start;
230
231 auto s = primary_start;
232 for (int thread = 0; thread < my_nthreads; ++thread) {
233 while (s < primary_end && my_indices[s] < secondary_ends[thread]) {
234 ++s;
235 }
236 my_secondary_nonzero_starts[thread + 1][c] = s;
237 }
238 }
239 }
