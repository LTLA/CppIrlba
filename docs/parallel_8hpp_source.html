187 void fragment_threads() {
188 auto total_nzeros = my_ptrs[my_primary_dim]; // last element - not using back() to avoid an extra requirement on PointerArray.
189 PointerType per_thread = (total_nzeros / my_nthreads) + (total_nzeros % my_nthreads > 0); // i.e., ceiling.
190
191 // Splitting columns across threads so each thread processes the same number of nonzero elements.
192 my_primary_starts.resize(my_nthreads);
193 my_primary_ends.resize(my_nthreads);
194 {
195 Eigen::Index primary_counter = 0;
196 PointerType sofar = per_thread;
197 for (int t = 0; t < my_nthreads; ++t) {
198 my_primary_starts[t] = primary_counter;
199 while (primary_counter < my_primary_dim && my_ptrs[primary_counter + 1] <= sofar) {
200 ++primary_counter;
201 }
202 my_primary_ends[t] = primary_counter;
203 sofar += per_thread;
204 }
205 }
206
207 // Splitting rows across threads so each thread processes the same number of nonzero elements.
208 my_secondary_nonzero_starts.resize(my_nthreads + 1, std::vector<PointerType>(my_primary_dim));
209 {
210 std::vector<PointerType> secondary_nonzeros(my_secondary_dim);
211 for (PointerType i = 0; i < total_nzeros; ++i) { // don't using range for loop to avoid an extra requirement on IndexArray.
212 ++(secondary_nonzeros[my_indices[i]]);
213 }
214
215 std::vector<IndexType> secondary_ends(my_nthreads);
216 IndexType secondary_counter = 0;
217 PointerType sofar = per_thread;
218 PointerType cum_rows = 0;
219
220 for (int t = 0; t < my_nthreads; ++t) {
221 while (secondary_counter < my_secondary_dim && cum_rows <= sofar) {
222 cum_rows += secondary_nonzeros[secondary_counter];
223 ++secondary_counter;
224 }
225 secondary_ends[t] = secondary_counter;
226 sofar += per_thread;
227 }
228
229 for (Eigen::Index c = 0; c < my_primary_dim; ++c) {
230 auto primary_start = my_ptrs[c], primary_end = my_ptrs[c + 1];
231 my_secondary_nonzero_starts[0][c] = primary_start;
232
233 auto s = primary_start;
234 for (int thread = 0; thread < my_nthreads; ++thread) {
235 while (s < primary_end && my_indices[s] < secondary_ends[thread]) {
236 ++s;
237 }
238 my_secondary_nonzero_starts[thread + 1][c] = s;
239 }
240 }
241 }
