35 ParallelSparseMatrixCore(
36 Eigen::Index nrow,
37 Eigen::Index ncol,
38 ValueArray_ x,
39 IndexArray_ i,
40 PointerArray_ p,
41 bool column_major,
42 int num_threads
43 ) :
44 my_primary_dim(column_major ? ncol : nrow),
45 my_secondary_dim(column_major ? nrow : ncol),
46 my_num_threads(num_threads),
47 my_values(std::move(x)),
48 my_indices(std::move(i)),
49 my_ptrs(std::move(p)),
50 my_column_major(column_major)
51 {
52 if (num_threads > 1) {
53 const auto total_nzeros = my_ptrs[my_primary_dim]; // last element - not using back() to avoid an extra requirement on PointerArray.
54 PointerType per_thread = (total_nzeros / my_num_threads) + (total_nzeros % my_num_threads > 0); // i.e., ceiling.
55
56 // Splitting columns across threads so each thread processes the same number of nonzero elements.
57 my_primary_starts.resize(my_num_threads);
58 my_primary_ends.resize(my_num_threads);
59 {
60 Eigen::Index primary_counter = 0;
61 PointerType sofar = per_thread;
62 for (int t = 0; t < my_num_threads; ++t) {
63 my_primary_starts[t] = primary_counter;
64 while (primary_counter < my_primary_dim && my_ptrs[primary_counter + 1] <= sofar) {
65 ++primary_counter;
66 }
67 my_primary_ends[t] = primary_counter;
68 sofar += per_thread;
69 }
70 }
71
72 // Splitting rows across threads so each thread processes the same number of nonzero elements.
73 my_secondary_ends.resize(my_num_threads + 1);
74 my_secondary_nonzero_starts.resize(my_num_threads + 1, std::vector<PointerType>(my_primary_dim));
75 {
76 std::vector<PointerType> secondary_nonzeros(my_secondary_dim);
77 for (PointerType i = 0; i < total_nzeros; ++i) { // don't using range for loop to avoid an extra requirement on IndexArray.
78 ++(secondary_nonzeros[my_indices[i]]);
79 }
80
81 IndexType secondary_counter = 0;
82 PointerType sofar = per_thread;
83 PointerType cum_rows = 0;
84
85 for (int t = 0; t < my_num_threads; ++t) {
86 while (secondary_counter < my_secondary_dim && cum_rows <= sofar) {
87 cum_rows += secondary_nonzeros[secondary_counter];
88 ++secondary_counter;
89 }
90 my_secondary_ends[t + 1] = secondary_counter;
91 sofar += per_thread;
92 }
93
94 for (Eigen::Index c = 0; c < my_primary_dim; ++c) {
95 auto primary_start = my_ptrs[c], primary_end = my_ptrs[c + 1];
96 my_secondary_nonzero_starts[0][c] = primary_start;
97
98 auto s = primary_start;
99 for (int thread = 0; thread < my_num_threads; ++thread) {
100 auto limit = my_secondary_ends[thread + 1];
101 while (s < primary_end && my_indices[s] < limit) {
102 ++s;
103 }
104 my_secondary_nonzero_starts[thread + 1][c] = s;
105 }
106 }
107 }
108 }
109 }

174 template<typename EigenVector_>
175 void indirect_multiply(const EigenVector_& rhs, std::vector<std::vector<typename EigenVector_::Scalar> >& thread_buffers, EigenVector_& output) const {
176 if (my_num_threads == 1) {
177 output.setZero();
178 for (Eigen::Index c = 0; c < my_primary_dim; ++c) {
179 auto start = my_ptrs[c];
180 auto end = my_ptrs[c + 1];
181 auto val = rhs.coeff(c);
182 for (PointerType s = start; s < end; ++s) {
183 output.coeffRef(my_indices[s]) += my_values[s] * val;
184 }
185 }
186 return;
187 }
188
189 parallelize(my_num_threads, [&](int t) -> void {
190 auto secondary_start = my_secondary_ends[t];
191 auto secondary_end = my_secondary_ends[t + 1];
192 size_t secondary_len = secondary_end - secondary_start;
193
194 // Using a separate buffer for the other threads to avoid false
195 // sharing. On first use, each buffer is allocated within each
196 // thread to give malloc a chance of using thread-specific arenas.
197 typename EigenVector_::Scalar* optr;
198 if (t != 0) {
199 auto& curbuffer = thread_buffers[t - 1];
200 curbuffer.resize(secondary_len);
201 optr = curbuffer.data();
202 } else {
203 optr = output.data() + secondary_start;
204 }
205 std::fill_n(optr, secondary_len, static_cast<typename EigenVector_::Scalar>(0));
206
207 const auto& nz_starts = my_secondary_nonzero_starts[t];
208 const auto& nz_ends = my_secondary_nonzero_starts[t + 1];
209 for (Eigen::Index c = 0; c < my_primary_dim; ++c) {
210 auto nz_start = nz_starts[c];
211 auto nz_end = nz_ends[c];
212 auto val = rhs.coeff(c);
213 for (PointerType s = nz_start; s < nz_end; ++s) {
214 optr[my_indices[s] - secondary_start] += my_values[s] * val;
215 }
216 }
217
218 if (t != 0) {
219 std::copy_n(optr, secondary_len, output.data() + secondary_start);
220 }
221 });
222
223 return;
224 }

227 template<typename EigenVector_>
228 void direct_multiply(const EigenVector_& rhs, EigenVector_& output) const {
229 if (my_num_threads == 1) {
230 for (Eigen::Index c = 0; c < my_primary_dim; ++c) {
231 output.coeffRef(c) = column_dot_product<typename EigenVector_::Scalar>(c, rhs);
232 }
233 return;
234 }
235
236 parallelize(my_num_threads, [&](int t) -> void {
237 auto curstart = my_primary_starts[t];
238 auto curend = my_primary_ends[t];
239 for (size_t c = curstart; c < curend; ++c) {
240 output.coeffRef(c) = column_dot_product<typename EigenVector_::Scalar>(c, rhs);
241 }
242 });
243
244 return;
245 }
246
247private:
248 template<typename Scalar_, class EigenVector_>
249 Scalar_ column_dot_product(std::size_t c, const EigenVector_& rhs) const {
250 PointerType primary_start = my_ptrs[c], primary_end = my_ptrs[c + 1];
251 Scalar_ dot = 0;
252 for (PointerType s = primary_start; s < primary_end; ++s) {
253 dot += my_values[s] * rhs.coeff(my_indices[s]);
254 }
255 return dot;
