83template<class Matrix_, class EigenMatrix_, class EigenVector_>
84std::pair<bool, int> compute(const Matrix_& matrix, Eigen::Index number, EigenMatrix_& outU, EigenMatrix_& outV, EigenVector_& outD, const Options& options) {
85 Eigen::Index smaller = std::min(matrix.rows(), matrix.cols());
86 Eigen::Index requested_number = number;
87 if (requested_number > smaller) {
88 if (options.cap_number) {
89 requested_number = smaller;
90 } else {
91 throw std::runtime_error("requested number of singular values cannot be greater than the smaller matrix dimension");
92 }
93 } else if (requested_number == smaller && !options.exact_for_large_number) {
94 throw std::runtime_error("requested number of singular values must be less than the smaller matrix dimension for IRLBA iterations");
95 }
96
97 // Falling back to an exact SVD for small matrices or if the requested number is too large
98 // (not enough of a workspace). Hey, I don't make the rules.
99 if ((options.exact_for_small_matrix && smaller < 6) || (options.exact_for_large_number && requested_number * 2 >= smaller)) {
100 internal::exact(matrix, requested_number, outU, outV, outD);
101 return std::make_pair(true, 0);
102 }
103
104 Eigen::Index work = std::min(requested_number + options.extra_work, smaller);
105
106 EigenMatrix_ V(matrix.cols(), work);
107 std::mt19937_64 eng(options.seed);
108 if (options.initial) {
109 auto init = reinterpret_cast<EigenVector_*>(options.initial);
110 if (init->size() != V.rows()) {
111 throw std::runtime_error("initialization vector does not have expected number of rows");
112 }
113 V.col(0) = *init;
114 } else {
115 internal::fill_with_random_normals(V, 0, eng);
116 }
117 V.col(0) /= V.col(0).norm();
118
119 bool converged = false;
120 int iter = 0;
121 Eigen::Index k = 0;
122 Eigen::JacobiSVD<EigenMatrix_> svd(work, work, Eigen::ComputeThinU | Eigen::ComputeThinV);
123
124 internal::LanczosWorkspace<EigenVector_, Matrix_> lptmp(matrix);
125
126 EigenMatrix_ W(matrix.rows(), work);
127 EigenMatrix_ Wtmp(matrix.rows(), work);
128 EigenMatrix_ Vtmp(matrix.cols(), work);
129
130 EigenMatrix_ B(work, work);
131 B.setZero(work, work);
132 EigenVector_ res(work);
133 EigenVector_ F(matrix.cols());
134
135 EigenVector_ prevS(work);
136 typename EigenMatrix_::Scalar svtol = options.singular_value_ratio_tolerance;
137 typename EigenMatrix_::Scalar tol = options.convergence_tolerance;
138 typename EigenMatrix_::Scalar svtol_actual = (svtol >= 0 ? svtol : tol);
139
140 for (; iter < options.max_iterations; ++iter) {
141 // Technically, this is only a 'true' Lanczos bidiagonalization
142 // when k = 0. All other times, we're just recycling the machinery,
143 // see the text below Equation 3.11 in Baglama and Reichel.
144 internal::run_lanczos_bidiagonalization(matrix, W, V, B, eng, lptmp, k, options);
145
146// if (iter < 2) {
147// std::cout << "B is currently:\n" << B << std::endl;
148// std::cout << "W is currently:\n" << W << std::endl;
149// std::cout << "V is currently:\n" << V << std::endl;
150// }
151
152 svd.compute(B);
153 const auto& BS = svd.singularValues();
154 const auto& BU = svd.matrixU();
155 const auto& BV = svd.matrixV();
156
157 // Checking for convergence.
158 if (B(work - 1, work - 1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
159 converged = true;
160 break;
161 }
162
163 const auto& F = lptmp.F; // i.e., the residuals, see Algorithm 2.1 of Baglama and Reichel.
164 auto R_F = F.norm();
165
166 // Computes the convergence criterion defined in on the LHS of
167 // Equation 2.13 of Baglama and Riechel. (e_m is literally the unit
168 // vector along the m-th dimension, so it ends up just being the
169 // m-th row of the U matrix.) We expose it here as we will be
170 // re-using the same values to update B, see below.
171 res = R_F * BU.row(work - 1);
172
173 Eigen::Index n_converged = 0;
174 if (iter > 0) {
175 auto Smax = *std::max_element(BS.begin(), BS.end());
176 auto threshold = Smax * tol;
177
178 for (Eigen::Index j = 0; j < work; ++j) {
179 if (std::abs(res[j]) <= threshold) {
180 auto ratio = std::abs(BS[j] - prevS[j]) / BS[j];
181 if (ratio <= svtol_actual) {
182 ++n_converged;
183 }
184 }
185 }
186
187 if (n_converged >= requested_number) {
188 converged = true;
189 break;
190 }
191 }
192 prevS = BS;
193
194 // Setting 'k'. This looks kinda weird, but this is deliberate,
195 // see the text below Algorithm 3.1 of Baglama and Reichel.
196 //
197 // - Our n_converged is their k'.
198 // - Our requested_number is their k.
199 // - Our work is their m.
200 // - Our k is their k + k''.
201 if (n_converged + requested_number > k) {
202 k = n_converged + requested_number;
203 }
204 if (k > work - 3) {
205 k = work - 3;
206 }
207 if (k < 1) {
208 k = 1;
209 }
210
211 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
212 V.leftCols(k) = Vtmp.leftCols(k);
213
214 // See Equation 3.2 of Baglama and Reichel, where our 'V' is their
215 // 'P', and our 'F / R_F' is their 'p_{m+1}' (Equation 2.2). 'F'
216 // was orthogonal to the old 'V' and so it should still be
217 // orthogonal to the new left-most columns of 'V'; the input
218 // expectations of the Lanczos bidiagonalization are still met, so
219 // V is ok to use in the next run_lanczos_bidiagonalization().
220 V.col(k) = F / R_F;
221
222 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
223 W.leftCols(k) = Wtmp.leftCols(k);
224
225 B.setZero(work, work);
226 for (Eigen::Index l = 0; l < k; ++l) {
227 B(l, l) = BS[l];
228
229 // This assignment looks weird but is deliberate, see Equation
230 // 3.6 of Baglama and Reichel. Happily, this is the same value
231 // used to determine convergence in 2.13, so we just re-use it.
232 // (See the equation just above Equation 3.5; I think they
233 // misplaced a tilde on the final 'u', given no other 'u' has
234 // a B_m superscript as well as a tilde.)
235 B(l, k) = res[l];
236 }
237 }
238
239 // See Equation 2.11 of Baglama and Reichel for how to get from B's
240 // singular triplets to mat's singular triplets.
241 outD.resize(requested_number);
242 outD = svd.singularValues().head(requested_number);
243
244 outU.resize(matrix.rows(), requested_number);
245 outU.noalias() = W * svd.matrixU().leftCols(requested_number);
246
247 outV.resize(matrix.cols(), requested_number);
248 outV.noalias() = V * svd.matrixV().leftCols(requested_number);
249
250 return std::make_pair(converged, iter + 1);
251}

342template<class Matrix_, class EigenMatrix_, class EigenVector_>
343std::pair<bool, int> compute(const Matrix_& matrix, bool center, bool scale, Eigen::Index number, EigenMatrix_& outU, EigenMatrix_& outV, EigenVector_& outD, const Options& options) {
344 if (!scale && !center) {
345 return compute(matrix, number, outU, outV, outD, options);
346 }
347
348 auto nr = matrix.rows();
349 auto nc = matrix.cols();
350
351 Eigen::VectorXd center0;
352 if (center) {
353 if (nr < 1) {
354 throw std::runtime_error("cannot center with no observations");
355 }
356 center0.resize(nc);
357 }
358
359 Eigen::VectorXd scale0;
360 if (scale) {
361 if (nr < 2) {
362 throw std::runtime_error("cannot scale with fewer than two observations");
363 }
364 scale0.resize(nc);
365 }
366
367 for (Eigen::Index i = 0; i < nc; ++i) {
368 double mean = 0;
369 if (center) {
370 mean = matrix.col(i).sum() / nr;
371 center0[i] = mean;
372 }
373 if (scale) {
374 EigenVector_ current = matrix.col(i); // force it to be a Vector, even if it's a sparse matrix.
375 typename EigenMatrix_::Scalar var = 0;
376 for (auto x : current) {
377 var += (x - mean)*(x - mean);
378 }
379
380 if (var) {
381 scale0[i] = std::sqrt(var/(nr - 1));
382 } else {
383 scale0[i] = 1;
384 }
385 }
386 }
387
388 if (center) {
389 Centered<EigenMatrix_, EigenVector_> centered(matrix, center0);
390 if (scale) {
391 auto centered_scaled = make_Scaled<true>(centered, scale0, true);
392 return compute(centered_scaled, number, outU, outV, outD, options);
393 } else {
394 return compute(centered, number, outU, outV, outD, options);
395 }
396 } else {
397 auto scaled = make_Scaled<true>(matrix, scale0, true);
398 return compute(scaled, number, outU, outV, outD, options);
399 }
400}
