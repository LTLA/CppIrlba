190 std::pair<bool, int> run(
191 const M& mat,
192 Eigen::MatrixXd& outU,
193 Eigen::MatrixXd& outV,
194 Eigen::VectorXd& outD,
195 Engine* eng = null_rng())
196 {
197 Eigen::VectorXd center0, scale0;
198
199 if constexpr(SCALE || CENTER) {
200 if constexpr(SCALE) {
201 assert(mat.rows() >= 2); // avoid div by zero.
202 scale0.resize(mat.cols());
203 }
204 if constexpr(CENTER) {
205 assert(mat.rows() >= 1);
206 center0.resize(mat.cols());
207 }
208
209 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
210 double mean = 0;
211 if constexpr(CENTER) {
212 mean = mat.col(i).sum() / mat.rows();
213 center0[i] = mean;
214 }
215 if constexpr(SCALE) {
216 Eigen::VectorXd current = mat.col(i); // force it to be a VectorXd, even if it's a sparse matrix.
217 double var = 0;
218 for (auto x : current) {
219 var += (x - mean)*(x - mean);
220 }
221 scale0[i] = std::sqrt(var/(mat.rows() - 1));
222 }
223 }
224 }
225
226 if constexpr(CENTER) {
227 if constexpr(SCALE) {
228 return run_internal(mat, center0, scale0, outU, outV, outD, eng);
229 } else {
230 return run_internal(mat, center0, false, outU, outV, outD, eng);
231 }
232 } else {
233 if constexpr(SCALE) {
234 return run_internal(mat, false, scale0, outU, outV, outD, eng);
235 } else {
236 return run_internal(mat, false, false, outU, outV, outD, eng);
237 }

294 template<class M, class CENTER, class SCALE, class Engine>
295 std::pair<bool, int> run_internal(const M& mat, const CENTER& center, const SCALE& scale, Engine& eng, Eigen::MatrixXd& outU, Eigen::MatrixXd& outV, Eigen::VectorXd& outD) {
296 const int smaller = std::min(mat.rows(), mat.cols());
297 assert(number < smaller);
298
299 // Falling back to an exact SVD for small matrices.
300 if (smaller < 6) {
301 exact(mat, center, scale, outU, outV, outD);
302 return std::make_pair(true, 0);
303 }
304
305 const int work = std::min(number + extra_work, smaller);
306
307 W.resize(mat.rows(), work);
308 Wtmp.resize(mat.rows(), work);
309 V.resize(mat.cols(), work);
310 Vtmp.resize(mat.cols(), work);
311 res.resize(work);
312 F.resize(mat.cols());
313
314 if (initV.size()) {
315 assert(initV.size() == V.rows());
316 V.col(0) = initV;
317 } else {
318 fill_with_random_normals(V, 0, eng);
319 }
320 V.col(0) /= V.col(0).norm();
321
322 B.resize(work, work);
323 B.setZero(work, work);
324
325 bool converged = false;
326 int iter = 0, k =0;
327 Eigen::JacobiSVD<Eigen::MatrixXd> svd(work, work, Eigen::ComputeThinU | Eigen::ComputeThinV);
328
329 for (; iter < maxit; ++iter) {
330 // Technically, this is only a 'true' Lanczos bidiagonalization
331 // when k = 0. All other times, we're just recycling the machinery,
332 // see the text below Equation 3.11 in Baglama and Reichel.
333 lp.run(mat, W, V, B, center, scale, eng, k);
334
335// if (iter < 2) {
336// std::cout << "B is currently:\n" << B << std::endl;
337// std::cout << "W is currently:\n" << W << std::endl;
338// std::cout << "V is currently:\n" << V << std::endl;
339// }
340
341 svd.compute(B);
342 const auto& BS = svd.singularValues();
343 const auto& BU = svd.matrixU();
344 const auto& BV = svd.matrixV();
345
346 // Checking for convergence.
347 if (B(work-1, work-1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
348 converged = true;
349 break;
350 }
351
352 F = lp.residuals();
353 double R_F = F.norm();
354 F /= R_F;
355
356 // Computes the convergence criterion defined in on the LHS of Equation 2.13 of Baglama and Riechel.
357 // We expose it here as we will be re-using the same values to update B, see below.
358 res = R_F * BU.row(work - 1);
359
360 int n_converged = 0;
361 if (iter > 0) {
362 n_converged = convtest.run(BS, res);
363 if (n_converged >= number) {
364 converged = true;
365 break;
366 }
367 }
368 convtest.set_last(BS);
369
370 // Setting 'k'. This looks kinda weird, but this is deliberate,
371 // see the text below Algorithm 3.1 of Baglama and Reichel.
372 if (n_converged + number > k) {
373 k = n_converged + number;
374 }
375 if (k > work - 3) {
376 k = work - 3;
377 }
378 if (k < 1) {
379 k = 1;
380 }
381
382 // Updating B, W and V.
383 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
384 V.leftCols(k) = Vtmp.leftCols(k);
385
386 V.col(k) = F; // See Equation 3.2 of Baglama and Reichel, where our 'V' is
387 // their 'P', and our 'F' is their 'p_{m+1}' (2.2).
388 // 'F' was orthogonal to the old 'V' and it so it
389 // should still be orthogonal to the new left-most
390 // columns of 'V'; the input expectations of 'lp'
391 // are still met.
392
393 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
394 W.leftCols(k) = Wtmp.leftCols(k);
395
396 B.setZero(work, work);
397 for (int l = 0; l < k; ++l) {
398 B(l, l) = BS[l];
399 B(l, k) = res[l]; // this looks weird but is deliberate, see
400 // Equation 3.6 of Baglama and Reichel.
401 // By happy coincidence, this is the same
402 // value used to determine convergence in
403 // 2.13, so we can just re-use it.
404 }
405 }
406
407 // See Equation 2.11 of Baglama and Reichel for how to get from B's
408 // singular triplets to mat's singular triplets.
409 outD.resize(number);
410 outD = svd.singularValues().head(number);
411
412 outU.resize(mat.rows(), number);
413 outU.noalias() = W * svd.matrixU().leftCols(number);
414
415 outV.resize(mat.cols(), number);
416 outV.noalias() = V * svd.matrixV().leftCols(number);
417
418 return std::make_pair(converged, iter + 1);
419 }

423 void exact(const M& mat, const CENTER& center, const SCALE& scale, Eigen::MatrixXd& outU, Eigen::MatrixXd& outV, Eigen::VectorXd& outD) {
424 Eigen::BDCSVD<Eigen::MatrixXd> svd(mat.rows(), mat.cols(), Eigen::ComputeThinU | Eigen::ComputeThinV);
425 constexpr bool do_center = !std::is_same<CENTER, bool>::value;
426 constexpr bool do_scale = !std::is_same<SCALE, bool>::value;
427
428 if constexpr(std::is_same<M, Eigen::MatrixXd>::value && !do_center && !do_scale) {
429 svd.compute(mat);
430 } else {
431 Eigen::MatrixXd adjusted(mat);
432
433 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
434 if constexpr(do_center) {
435 for (Eigen::Index j = 0; j < mat.rows(); ++j) {
436 adjusted(j, i) -= center[i];
437 }
438 }
439 if constexpr(do_scale) {
440 adjusted.col(i) /= scale[i];
441 }
442 }
443 svd.compute(adjusted);
444 }
445
446 outD.resize(number);
447 outD = svd.singularValues().head(number);
448
449 outU.resize(mat.rows(), number);
450 outU = svd.matrixU().leftCols(number);
451
452 outV.resize(mat.cols(), number);
453 outV = svd.matrixV().leftCols(number);
454
455 return;
456 }
