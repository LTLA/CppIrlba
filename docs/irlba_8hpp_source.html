186 std::pair<bool, int> run(
187 const M& mat,
188 Eigen::MatrixXd& outU,
189 Eigen::MatrixXd& outV,
190 Eigen::VectorXd& outD,
191 Engine* eng = null_rng(),
192 Eigen::VectorXd* init = NULL)
193 {
194 Eigen::VectorXd center0, scale0;
195
196 if constexpr(SCALE || CENTER) {
197 if constexpr(SCALE) {
198 assert(mat.rows() >= 2); // avoid div by zero.
199 scale0.resize(mat.cols());
200 }
201 if constexpr(CENTER) {
202 assert(mat.rows() >= 1);
203 center0.resize(mat.cols());
204 }
205
206 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
207 double mean = 0;
208 if constexpr(CENTER) {
209 mean = mat.col(i).sum() / mat.rows();
210 center0[i] = mean;
211 }
212 if constexpr(SCALE) {
213 Eigen::VectorXd current = mat.col(i); // force it to be a VectorXd, even if it's a sparse matrix.
214 double var = 0;
215 for (auto x : current) {
216 var += (x - mean)*(x - mean);
217 }
218 scale0[i] = std::sqrt(var/(mat.rows() - 1));
219 }
220 }
221 }
222
223 if constexpr(CENTER) {
224 if constexpr(SCALE) {
225 return run_internal(mat, center0, scale0, outU, outV, outD, eng, init);
226 } else {
227 return run_internal(mat, center0, false, outU, outV, outD, eng, init);
228 }
229 } else {
230 if constexpr(SCALE) {
231 return run_internal(mat, false, scale0, outU, outV, outD, eng, init);
232 } else {
233 return run_internal(mat, false, false, outU, outV, outD, eng, init);
234 }

303 template<class M, class CENTER, class SCALE, class Engine>
304 std::pair<bool, int> run_internal(
305 const M& mat,
306 const CENTER& center,
307 const SCALE& scale,
308 Engine& eng,
309 Eigen::MatrixXd& outU,
310 Eigen::MatrixXd& outV,
311 Eigen::VectorXd& outD,
312 Eigen::VectorXd* init)
313 {
314 const int smaller = std::min(mat.rows(), mat.cols());
315 if (number >= smaller) {
316 throw std::runtime_error("requested number of singular values must be less than smaller matrix dimension");
317 }
318
319 // Falling back to an exact SVD for small matrices.
320 if (smaller < 6) {
321 exact(mat, center, scale, outU, outV, outD);
322 return std::make_pair(true, 0);
323 }
324
325 const int work = std::min(number + extra_work, smaller);
326
327 Eigen::MatrixXd V(mat.cols(), work);
328 if (init) {
329 if (init->size() != V.rows()) {
330 throw std::runtime_error("initialization vector does not have expected number of rows");
331 }
332 V.col(0) = *init;
333 } else {
334 fill_with_random_normals(V, 0, eng);
335 }
336 V.col(0) /= V.col(0).norm();
337
338 bool converged = false;
339 int iter = 0, k =0;
340 Eigen::JacobiSVD<Eigen::MatrixXd> svd(work, work, Eigen::ComputeThinU | Eigen::ComputeThinV);
341 auto lptmp = lp.initialize(mat);
342
343 Eigen::MatrixXd W(mat.rows(), work);
344 Eigen::MatrixXd Wtmp(mat.rows(), work);
345 Eigen::MatrixXd Vtmp(mat.cols(), work);
346
347 Eigen::MatrixXd B(work, work);
348 B.setZero(work, work);
349 Eigen::VectorXd res(work);
350 Eigen::VectorXd F(mat.cols());
351
352 Eigen::VectorXd prevS(work);
353
354 for (; iter < maxit; ++iter) {
355 // Technically, this is only a 'true' Lanczos bidiagonalization
356 // when k = 0. All other times, we're just recycling the machinery,
357 // see the text below Equation 3.11 in Baglama and Reichel.
358 lp.run(mat, W, V, B, center, scale, eng, lptmp, k);
359
360// if (iter < 2) {
361// std::cout << "B is currently:\n" << B << std::endl;
362// std::cout << "W is currently:\n" << W << std::endl;
363// std::cout << "V is currently:\n" << V << std::endl;
364// }
365
366 svd.compute(B);
367 const auto& BS = svd.singularValues();
368 const auto& BU = svd.matrixU();
369 const auto& BV = svd.matrixV();
370
371 // Checking for convergence.
372 if (B(work-1, work-1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
373 converged = true;
374 break;
375 }
376
377 F = lptmp.residuals();
378 double R_F = F.norm();
379 F /= R_F;
380
381 // Computes the convergence criterion defined in on the LHS of Equation 2.13 of Baglama and Riechel.
382 // We expose it here as we will be re-using the same values to update B, see below.
383 res = R_F * BU.row(work - 1);
384
385 int n_converged = 0;
386 if (iter > 0) {
387 n_converged = convtest.run(BS, res, prevS);
388 if (n_converged >= number) {
389 converged = true;
390 break;
391 }
392 }
393 prevS = BS;
394
395 // Setting 'k'. This looks kinda weird, but this is deliberate,
396 // see the text below Algorithm 3.1 of Baglama and Reichel.
397 if (n_converged + number > k) {
398 k = n_converged + number;
399 }
400 if (k > work - 3) {
401 k = work - 3;
402 }
403 if (k < 1) {
404 k = 1;
405 }
406
407 // Updating B, W and V.
408 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
409 V.leftCols(k) = Vtmp.leftCols(k);
410
411 V.col(k) = F; // See Equation 3.2 of Baglama and Reichel, where our 'V' is
412 // their 'P', and our 'F' is their 'p_{m+1}' (2.2).
413 // 'F' was orthogonal to the old 'V' and it so it
414 // should still be orthogonal to the new left-most
415 // columns of 'V'; the input expectations of 'lp'
416 // are still met.
417
418 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
419 W.leftCols(k) = Wtmp.leftCols(k);
420
421 B.setZero(work, work);
422 for (int l = 0; l < k; ++l) {
423 B(l, l) = BS[l];
424 B(l, k) = res[l]; // this looks weird but is deliberate, see
425 // Equation 3.6 of Baglama and Reichel.
426 // By happy coincidence, this is the same
427 // value used to determine convergence in
428 // 2.13, so we can just re-use it.
429 }
430 }
431
432 // See Equation 2.11 of Baglama and Reichel for how to get from B's
433 // singular triplets to mat's singular triplets.
434 outD.resize(number);
435 outD = svd.singularValues().head(number);
436
437 outU.resize(mat.rows(), number);
438 outU.noalias() = W * svd.matrixU().leftCols(number);
439
440 outV.resize(mat.cols(), number);
441 outV.noalias() = V * svd.matrixV().leftCols(number);
442
443 return std::make_pair(converged, iter + 1);
444 }

447 template<class M, class CENTER, class SCALE>
448 void exact(const M& mat, const CENTER& center, const SCALE& scale, Eigen::MatrixXd& outU, Eigen::MatrixXd& outV, Eigen::VectorXd& outD) {
449 Eigen::BDCSVD<Eigen::MatrixXd> svd(mat.rows(), mat.cols(), Eigen::ComputeThinU | Eigen::ComputeThinV);
450 constexpr bool do_center = !std::is_same<CENTER, bool>::value;
451 constexpr bool do_scale = !std::is_same<SCALE, bool>::value;
452
453 if constexpr(std::is_same<M, Eigen::MatrixXd>::value && !do_center && !do_scale) {
454 svd.compute(mat);
455 } else {
456 auto compute = [&](Eigen::MatrixXd& adjusted) -> void {
457 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
458 if constexpr(do_center) {
459 for (Eigen::Index j = 0; j < mat.rows(); ++j) {
460 adjusted(j, i) -= center[i];
461 }
462 }
463 if constexpr(do_scale) {
464 adjusted.col(i) /= scale[i];
465 }
466 }
467 svd.compute(adjusted);
468 };
469
470 if constexpr(has_realize_method<M>::value) {
471 Eigen::MatrixXd adjusted = mat.realize();
472 compute(adjusted);
473 } else {
474 Eigen::MatrixXd adjusted(mat);
475 compute(adjusted);
476 }
477 }
478
479 outD.resize(number);
480 outD = svd.singularValues().head(number);
481
482 outU.resize(mat.rows(), number);
483 outU = svd.matrixU().leftCols(number);
484
485 outV.resize(mat.cols(), number);
486 outV = svd.matrixV().leftCols(number);
487
488 return;
489 }
