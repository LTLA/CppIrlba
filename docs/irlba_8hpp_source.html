187 template<class M, class Engine = std::mt19937_64>
188 std::pair<bool, int> run(
189 const M& mat,
190 bool center,
191 bool scale,
192 Eigen::MatrixXd& outU,
193 Eigen::MatrixXd& outV,
194 Eigen::VectorXd& outD,
195 Engine* eng = null_rng(),
196 Eigen::VectorXd* init = NULL)
197 const {
198 if (scale || center) {
199 Eigen::VectorXd center0, scale0;
200
201 if (center) {
202 if (mat.rows() < 1) {
203 throw std::runtime_error("cannot center with no observations");
204 }
205 center0.resize(mat.cols());
206 }
207
208 if (scale) {
209 if (mat.rows() < 2) {
210 throw std::runtime_error("cannot scale with fewer than two observations");
211 }
212 scale0.resize(mat.cols());
213 }
214
215 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
216 double mean = 0;
217 if (center) {
218 mean = mat.col(i).sum() / mat.rows();
219 center0[i] = mean;
220 }
221 if (scale) {
222 Eigen::VectorXd current = mat.col(i); // force it to be a VectorXd, even if it's a sparse matrix.
223 double var = 0;
224 for (auto x : current) {
225 var += (x - mean)*(x - mean);
226 }
227
228 if (var) {
229 scale0[i] = std::sqrt(var/(mat.rows() - 1));
230 } else {
231 scale0[i] = 1;
232 }
233 }
234 }
235
236 if (center) {
237 Centered<M> centered(&mat, &center0);
238 if (scale) {
239 Scaled<decltype(centered)> centered_scaled(&centered, &scale0);
240 return run(centered_scaled, outU, outV, outD, eng, init);
241 } else {
242 return run(centered, outU, outV, outD, eng, init);
243 }
244 } else {
245 Scaled<M> scaled(&mat, &scale0);
246 return run(scaled, outU, outV, outD, eng, init);
247 }
248 } else {
249 return run(mat, outU, outV, outD, eng, init);
250 }
251 }

317 template<class M, class Engine>
318 std::pair<bool, int> run_internal(
319 const M& mat,
320 Engine& eng,
321 Eigen::MatrixXd& outU,
322 Eigen::MatrixXd& outV,
323 Eigen::VectorXd& outD,
324 Eigen::VectorXd* init)
325 const {
326 const int smaller = std::min(mat.rows(), mat.cols());
327 if (number >= smaller) {
328 throw std::runtime_error("requested number of singular values must be less than smaller matrix dimension");
329 }
330
331 // Falling back to an exact SVD for small matrices.
332 if (smaller < 6) {
333 exact(mat, outU, outV, outD);
334 return std::make_pair(true, 0);
335 }
336
337 const int work = std::min(number + extra_work, smaller);
338
339 Eigen::MatrixXd V(mat.cols(), work);
340 if (init) {
341 if (init->size() != V.rows()) {
342 throw std::runtime_error("initialization vector does not have expected number of rows");
343 }
344 V.col(0) = *init;
345 } else {
346 fill_with_random_normals(V, 0, eng);
347 }
348 V.col(0) /= V.col(0).norm();
349
350 bool converged = false;
351 int iter = 0, k =0;
352 Eigen::JacobiSVD<Eigen::MatrixXd> svd(work, work, Eigen::ComputeThinU | Eigen::ComputeThinV);
353
354 auto lptmp = lp.initialize(mat);
355
356 Eigen::MatrixXd W(mat.rows(), work);
357 Eigen::MatrixXd Wtmp(mat.rows(), work);
358 Eigen::MatrixXd Vtmp(mat.cols(), work);
359
360 Eigen::MatrixXd B(work, work);
361 B.setZero(work, work);
362 Eigen::VectorXd res(work);
363 Eigen::VectorXd F(mat.cols());
364
365 Eigen::VectorXd prevS(work);
366
367 for (; iter < maxit; ++iter) {
368 // Technically, this is only a 'true' Lanczos bidiagonalization
369 // when k = 0. All other times, we're just recycling the machinery,
370 // see the text below Equation 3.11 in Baglama and Reichel.
371 lp.run(mat, W, V, B, eng, lptmp, k);
372
373// if (iter < 2) {
374// std::cout << "B is currently:\n" << B << std::endl;
375// std::cout << "W is currently:\n" << W << std::endl;
376// std::cout << "V is currently:\n" << V << std::endl;
377// }
378
379 svd.compute(B);
380 const auto& BS = svd.singularValues();
381 const auto& BU = svd.matrixU();
382 const auto& BV = svd.matrixV();
383
384 // Checking for convergence.
385 if (B(work-1, work-1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
386 converged = true;
387 break;
388 }
389
390 F = lptmp.residuals();
391 double R_F = F.norm();
392 F /= R_F;
393
394 // Computes the convergence criterion defined in on the LHS of Equation 2.13 of Baglama and Riechel.
395 // We expose it here as we will be re-using the same values to update B, see below.
396 res = R_F * BU.row(work - 1);
397
398 int n_converged = 0;
399 if (iter > 0) {
400 n_converged = convtest.run(BS, res, prevS);
401 if (n_converged >= number) {
402 converged = true;
403 break;
404 }
405 }
406 prevS = BS;
407
408 // Setting 'k'. This looks kinda weird, but this is deliberate,
409 // see the text below Algorithm 3.1 of Baglama and Reichel.
410 if (n_converged + number > k) {
411 k = n_converged + number;
412 }
413 if (k > work - 3) {
414 k = work - 3;
415 }
416 if (k < 1) {
417 k = 1;
418 }
419
420 // Updating B, W and V.
421 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
422 V.leftCols(k) = Vtmp.leftCols(k);
423
424 V.col(k) = F; // See Equation 3.2 of Baglama and Reichel, where our 'V' is
425 // their 'P', and our 'F' is their 'p_{m+1}' (2.2).
426 // 'F' was orthogonal to the old 'V' and it so it
427 // should still be orthogonal to the new left-most
428 // columns of 'V'; the input expectations of 'lp'
429 // are still met.
430
431 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
432 W.leftCols(k) = Wtmp.leftCols(k);
433
434 B.setZero(work, work);
435 for (int l = 0; l < k; ++l) {
436 B(l, l) = BS[l];
437 B(l, k) = res[l]; // this looks weird but is deliberate, see
438 // Equation 3.6 of Baglama and Reichel.
439 // By happy coincidence, this is the same
440 // value used to determine convergence in
441 // 2.13, so we can just re-use it.
442 }
443 }
444
445 // See Equation 2.11 of Baglama and Reichel for how to get from B's
446 // singular triplets to mat's singular triplets.
447 outD.resize(number);
448 outD = svd.singularValues().head(number);
449
450 outU.resize(mat.rows(), number);
451 outU.noalias() = W * svd.matrixU().leftCols(number);
452
453 outV.resize(mat.cols(), number);
454 outV.noalias() = V * svd.matrixV().leftCols(number);
455
456 return std::make_pair(converged, iter + 1);
457 }

460 template<class M>
461 void exact(const M& mat, Eigen::MatrixXd& outU, Eigen::MatrixXd& outV, Eigen::VectorXd& outD) const {
462 Eigen::BDCSVD<Eigen::MatrixXd> svd(mat.rows(), mat.cols(), Eigen::ComputeThinU | Eigen::ComputeThinV);
463
464 if constexpr(std::is_same<M, Eigen::MatrixXd>::value) {
465 svd.compute(mat);
466 } else {
467 if constexpr(has_realize_method<M>::value) {
468 Eigen::MatrixXd adjusted = mat.realize();
469 svd.compute(adjusted);
470 } else {
471 Eigen::MatrixXd adjusted(mat);
472 svd.compute(adjusted);
473 }
474 }
475
476 outD.resize(number);
477 outD = svd.singularValues().head(number);
478
479 outU.resize(mat.rows(), number);
480 outU = svd.matrixU().leftCols(number);
481
482 outV.resize(mat.cols(), number);
483 outV = svd.matrixV().leftCols(number);
484
485 return;
486 }
