239 template<class M, class Engine = std::mt19937_64>
240 std::pair<bool, int> run(
241 const M& mat,
242 bool center,
243 bool scale,
244 Eigen::MatrixXd& outU,
245 Eigen::MatrixXd& outV,
246 Eigen::VectorXd& outD,
247 Engine* eng = null_rng(),
248 Eigen::VectorXd* init = NULL)
249 const {
250 if (scale || center) {
251 Eigen::VectorXd center0, scale0;
252
253 if (center) {
254 if (mat.rows() < 1) {
255 throw std::runtime_error("cannot center with no observations");
256 }
257 center0.resize(mat.cols());
258 }
259
260 if (scale) {
261 if (mat.rows() < 2) {
262 throw std::runtime_error("cannot scale with fewer than two observations");
263 }
264 scale0.resize(mat.cols());
265 }
266
267 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
268 double mean = 0;
269 if (center) {
270 mean = mat.col(i).sum() / mat.rows();
271 center0[i] = mean;
272 }
273 if (scale) {
274 Eigen::VectorXd current = mat.col(i); // force it to be a VectorXd, even if it's a sparse matrix.
275 double var = 0;
276 for (auto x : current) {
277 var += (x - mean)*(x - mean);
278 }
279
280 if (var) {
281 scale0[i] = std::sqrt(var/(mat.rows() - 1));
282 } else {
283 scale0[i] = 1;
284 }
285 }
286 }
287
288 if (center) {
289 Centered<M> centered(&mat, &center0);
290 if (scale) {
291 Scaled<decltype(centered)> centered_scaled(&centered, &scale0);
292 return run(centered_scaled, outU, outV, outD, eng, init);
293 } else {
294 return run(centered, outU, outV, outD, eng, init);
295 }
296 } else {
297 Scaled<M> scaled(&mat, &scale0);
298 return run(scaled, outU, outV, outD, eng, init);
299 }
300 } else {
301 return run(mat, outU, outV, outD, eng, init);
302 }
303 }

352 template<class M, class Engine>
353 std::pair<bool, int> run_internal(
354 const M& mat,
355 Engine& eng,
356 Eigen::MatrixXd& outU,
357 Eigen::MatrixXd& outV,
358 Eigen::VectorXd& outD,
359 Eigen::VectorXd* init)
360 const {
361 const int smaller = std::min(mat.rows(), mat.cols());
362 int requested_number = number_;
363 if (requested_number > smaller) {
364 if (cap_number) {
365 requested_number = smaller;
366 } else {
367 throw std::runtime_error("requested number of singular values cannot be greater than the smaller matrix dimension");
368 }
369 } else if (requested_number == smaller && !exact_for_large_number) {
370 throw std::runtime_error("requested number of singular values must be less than the smaller matrix dimension for IRLBA iterations");
371 }
372
373 // Falling back to an exact SVD for small matrices or if the requested number is too large
374 // (not enough of a workspace). Hey, I don't make the rules.
375 if ((exact_for_small_matrix && smaller < 6) || (exact_for_large_number && requested_number * 2 >= smaller)) {
376 exact(mat, requested_number, outU, outV, outD);
377 return std::make_pair(true, 0);
378 }
379
380 const int work = std::min(requested_number + extra_work, smaller);
381
382 Eigen::MatrixXd V(mat.cols(), work);
383 if (init) {
384 if (init->size() != V.rows()) {
385 throw std::runtime_error("initialization vector does not have expected number of rows");
386 }
387 V.col(0) = *init;
388 } else {
389 fill_with_random_normals(V, 0, eng);
390 }
391 V.col(0) /= V.col(0).norm();
392
393 bool converged = false;
394 int iter = 0, k =0;
395 Eigen::JacobiSVD<Eigen::MatrixXd> svd(work, work, Eigen::ComputeThinU | Eigen::ComputeThinV);
396
397 auto lptmp = lp.initialize(mat);
398
399 Eigen::MatrixXd W(mat.rows(), work);
400 Eigen::MatrixXd Wtmp(mat.rows(), work);
401 Eigen::MatrixXd Vtmp(mat.cols(), work);
402
403 Eigen::MatrixXd B(work, work);
404 B.setZero(work, work);
405 Eigen::VectorXd res(work);
406 Eigen::VectorXd F(mat.cols());
407
408 Eigen::VectorXd prevS(work);
409
410 for (; iter < maxit; ++iter) {
411 // Technically, this is only a 'true' Lanczos bidiagonalization
412 // when k = 0. All other times, we're just recycling the machinery,
413 // see the text below Equation 3.11 in Baglama and Reichel.
414 lp.run(mat, W, V, B, eng, lptmp, k);
415
416// if (iter < 2) {
417// std::cout << "B is currently:\n" << B << std::endl;
418// std::cout << "W is currently:\n" << W << std::endl;
419// std::cout << "V is currently:\n" << V << std::endl;
420// }
421
422 svd.compute(B);
423 const auto& BS = svd.singularValues();
424 const auto& BU = svd.matrixU();
425 const auto& BV = svd.matrixV();
426
427 // Checking for convergence.
428 if (B(work-1, work-1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
429 converged = true;
430 break;
431 }
432
433 F = lptmp.residuals();
434 double R_F = F.norm();
435 F /= R_F;
436
437 // Computes the convergence criterion defined in on the LHS of Equation 2.13 of Baglama and Riechel.
438 // We expose it here as we will be re-using the same values to update B, see below.
439 res = R_F * BU.row(work - 1);
440
441 int n_converged = 0;
442 if (iter > 0) {
443 n_converged = convtest.run(BS, res, prevS);
444 if (n_converged >= requested_number) {
445 converged = true;
446 break;
447 }
448 }
449 prevS = BS;
450
451 // Setting 'k'. This looks kinda weird, but this is deliberate,
452 // see the text below Algorithm 3.1 of Baglama and Reichel.
453 if (n_converged + requested_number > k) {
454 k = n_converged + requested_number;
455 }
456 if (k > work - 3) {
457 k = work - 3;
458 }
459 if (k < 1) {
460 k = 1;
461 }
462
463 // Updating B, W and V.
464 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
465 V.leftCols(k) = Vtmp.leftCols(k);
466
467 V.col(k) = F; // See Equation 3.2 of Baglama and Reichel, where our 'V' is
468 // their 'P', and our 'F' is their 'p_{m+1}' (2.2).
469 // 'F' was orthogonal to the old 'V' and it so it
470 // should still be orthogonal to the new left-most
471 // columns of 'V'; the input expectations of 'lp'
472 // are still met.
473
474 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
475 W.leftCols(k) = Wtmp.leftCols(k);
476
477 B.setZero(work, work);
478 for (int l = 0; l < k; ++l) {
479 B(l, l) = BS[l];
480 B(l, k) = res[l]; // this looks weird but is deliberate, see
481 // Equation 3.6 of Baglama and Reichel.
482 // By happy coincidence, this is the same
483 // value used to determine convergence in
484 // 2.13, so we can just re-use it.
485 }
486 }
487
488 // See Equation 2.11 of Baglama and Reichel for how to get from B's
489 // singular triplets to mat's singular triplets.
490 outD.resize(requested_number);
491 outD = svd.singularValues().head(requested_number);
492
493 outU.resize(mat.rows(), requested_number);
494 outU.noalias() = W * svd.matrixU().leftCols(requested_number);
495
496 outV.resize(mat.cols(), requested_number);
497 outV.noalias() = V * svd.matrixV().leftCols(requested_number);
498
499 return std::make_pair(converged, iter + 1);
500 }

503 template<class M>
504 static void exact(const M& mat, int requested_number, Eigen::MatrixXd& outU, Eigen::MatrixXd& outV, Eigen::VectorXd& outD) {
505 Eigen::BDCSVD<Eigen::MatrixXd> svd(mat.rows(), mat.cols(), Eigen::ComputeThinU | Eigen::ComputeThinV);
506
507 if constexpr(std::is_same<M, Eigen::MatrixXd>::value) {
508 svd.compute(mat);
509 } else {
510 if constexpr(has_realize_method<M>::value) {
511 Eigen::MatrixXd adjusted = mat.realize();
512 svd.compute(adjusted);
513 } else {
514 Eigen::MatrixXd adjusted(mat);
515 svd.compute(adjusted);
516 }
517 }
518
519 outD.resize(requested_number);
520 outD = svd.singularValues().head(requested_number);
521
522 outU.resize(mat.rows(), requested_number);
523 outU = svd.matrixU().leftCols(requested_number);
524
525 outV.resize(mat.cols(), requested_number);
526 outV = svd.matrixV().leftCols(requested_number);
527
528 return;
529 }
