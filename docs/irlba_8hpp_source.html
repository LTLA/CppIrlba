174 Eigen::VectorXd center0, scale0;
175
176 if constexpr(SCALE || CENTER) {
177 if constexpr(SCALE) {
178 assert(mat.rows() >= 2); // avoid div by zero.
179 scale0.resize(mat.cols());
180 }
181 if constexpr(CENTER) {
182 assert(mat.rows() >= 1);
183 center0.resize(mat.cols());
184 }
185
186 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
187 double mean = 0;
188 if constexpr(CENTER) {
189 mean = mat.col(i).sum() / mat.rows();
190 center0[i] = mean;
191 }
192 if constexpr(SCALE) {
193 Eigen::VectorXd current = mat.col(i); // force it to be a VectorXd, even if it's a sparse matrix.
194 double var = 0;
195 for (auto x : current) {
196 var += (x - mean)*(x - mean);
197 }
198 scale0[i] = std::sqrt(var/(mat.rows() - 1));
199 }
200 }
201 }
202
203 if constexpr(CENTER) {
204 if constexpr(SCALE) {
205 return run_internal(mat, center0, scale0, norm, outU, outV, outD);
206 } else {
207 return run_internal(mat, center0, false, norm, outU, outV, outD);
208 }
209 } else {
210 if constexpr(SCALE) {
211 return run_internal(mat, false, scale0, norm, outU, outV, outD);
212 } else {
213 return run_internal(mat, false, false, norm, outU, outV, outD);
214 }
215 }

253 std::pair<bool, int> run_internal(const M& mat, const CENTER& center, const SCALE& scale, NORMSAMP& norm, Eigen::MatrixXd& outU, Eigen::MatrixXd& outV, Eigen::VectorXd& outD) {
254 const int smaller = std::min(mat.rows(), mat.cols());
255 assert(number < smaller);
256
257 // Falling back to an exact SVD for small matrices.
258 if (smaller < 6) {
259 exact(mat, center, scale, outU, outV, outD);
260 return std::make_pair(true, 0);
261 }
262
263 const int work = std::min(number + extra_work, smaller);
264
265 W.resize(mat.rows(), work);
266 Wtmp.resize(mat.rows(), work);
267 V.resize(mat.cols(), work);
268 Vtmp.resize(mat.cols(), work);
269 res.resize(work);
270 F.resize(mat.cols());
271
272 if (initV.size()) {
273 assert(initV.size() == V.rows());
274 V.col(0) = initV;
275 } else {
276 for (Eigen::Index i = 0; i < V.rows(); ++i) {
277 V(i, 0) = norm();
278 }
279 }
280 V.col(0) /= V.col(0).norm();
281
282 B.resize(work, work);
283 B.setZero(work, work);
284
285 bool converged = false;
286 int iter = 0, k =0;
287 Eigen::JacobiSVD<Eigen::MatrixXd> svd(work, work, Eigen::ComputeThinU | Eigen::ComputeThinV);
288
289 for (; iter < maxit; ++iter) {
290 // Technically, this is only a 'true' Lanczos bidiagonalization
291 // when k = 0. All other times, we're just recycling the machinery,
292 // see the text below Equation 3.11 in Baglama and Reichel.
293 lp.run(mat, W, V, B, center, scale, norm, k);
294
295// if (iter < 2) {
296// std::cout << "B is currently:\n" << B << std::endl;
297// std::cout << "W is currently:\n" << W << std::endl;
298// std::cout << "V is currently:\n" << V << std::endl;
299// }
300
301 svd.compute(B);
302 const auto& BS = svd.singularValues();
303 const auto& BU = svd.matrixU();
304 const auto& BV = svd.matrixV();
305
306 // Checking for convergence.
307 if (B(work-1, work-1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
308 converged = true;
309 break;
310 }
311
312 F = lp.residuals();
313 double R_F = F.norm();
314 F /= R_F;
315
316 // Computes the convergence criterion defined in on the LHS of Equation 2.13 of Baglama and Riechel.
317 // We expose it here as we will be re-using the same values to update B, see below.
318 res = R_F * BU.row(work - 1);
319
320 int n_converged = 0;
321 if (iter > 0) {
322 n_converged = convtest.run(BS, res);
323 if (n_converged >= number) {
324 converged = true;
325 break;
326 }
327 }
328 convtest.set_last(BS);
329
330 // Setting 'k'. This looks kinda weird, but this is deliberate,
331 // see the text below Algorithm 3.1 of Baglama and Reichel.
332 if (n_converged + number > k) {
333 k = n_converged + number;
334 }
335 if (k > work - 3) {
336 k = work - 3;
337 }
338 if (k < 1) {
339 k = 1;
340 }
341
342 // Updating B, W and V.
343 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
344 V.leftCols(k) = Vtmp.leftCols(k);
345
346 V.col(k) = F; // See Equation 3.2 of Baglama and Reichel, where our 'V' is
347 // their 'P', and our 'F' is their 'p_{m+1}' (2.2).
348 // 'F' was orthogonal to the old 'V' and it so it
349 // should still be orthogonal to the new left-most
350 // columns of 'V'; the input expectations of 'lp'
351 // are still met.
352
353 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
354 W.leftCols(k) = Wtmp.leftCols(k);
355
356 B.setZero(work, work);
357 for (int l = 0; l < k; ++l) {
358 B(l, l) = BS[l];
359 B(l, k) = res[l]; // this looks weird but is deliberate, see
360 // Equation 3.6 of Baglama and Reichel.
361 // By happy coincidence, this is the same
362 // value used to determine convergence in
363 // 2.13, so we can just re-use it.
364 }
365 }
366
367 // See Equation 2.11 of Baglama and Reichel for how to get from B's
368 // singular triplets to mat's singular triplets.
369 outD.resize(number);
370 outD = svd.singularValues().head(number);
371
372 outU.resize(mat.rows(), number);
373 outU.noalias() = W * svd.matrixU().leftCols(number);
374
375 outV.resize(mat.cols(), number);
376 outV.noalias() = V * svd.matrixV().leftCols(number);
377
378 return std::make_pair(converged, iter + 1);
379 }

382 void exact(const M& mat, const CENTER& center, const SCALE& scale, Eigen::MatrixXd& outU, Eigen::MatrixXd& outV, Eigen::VectorXd& outD) {
383 Eigen::BDCSVD<Eigen::MatrixXd> svd(mat.rows(), mat.cols(), Eigen::ComputeThinU | Eigen::ComputeThinV);
384 constexpr bool do_center = !std::is_same<CENTER, bool>::value;
385 constexpr bool do_scale = !std::is_same<SCALE, bool>::value;
386
387 if constexpr(std::is_same<M, Eigen::MatrixXd>::value && !do_center && !do_scale) {
388 svd.compute(mat);
389 } else {
390 Eigen::MatrixXd adjusted(mat);
391
392 for (Eigen::Index i = 0; i < mat.cols(); ++i) {
393 if constexpr(do_center) {
394 for (Eigen::Index j = 0; j < mat.rows(); ++j) {
395 adjusted(j, i) -= center[i];
396 }
397 }
398 if constexpr(do_scale) {
399 adjusted.col(i) /= scale[i];
400 }
401 }
402 svd.compute(adjusted);
403 }
404
405 outD.resize(number);
406 outD = svd.singularValues().head(number);
407
408 outU.resize(mat.rows(), number);
409 outU = svd.matrixU().leftCols(number);
410
411 outV.resize(mat.cols(), number);
412 outV = svd.matrixV().leftCols(number);
413
414 return;
415 }
