74template<class Matrix_, class EigenMatrix_, class EigenVector_>
75std::pair<bool, int> compute(
76 const Matrix_& matrix,
77 Eigen::Index number,
78 EigenMatrix_& outU,
79 EigenMatrix_& outV,
80 EigenVector_& outD,
81 const Options& options
82) {
83 Eigen::Index smaller = std::min(matrix.rows(), matrix.cols());
84 Eigen::Index requested_number = number;
85 if (requested_number > smaller) {
86 if (options.cap_number) {
87 requested_number = smaller;
88 } else {
89 throw std::runtime_error("requested number of singular values cannot be greater than the smaller matrix dimension");
90 }
91 } else if (requested_number == smaller && !options.exact_for_large_number) {
92 throw std::runtime_error("requested number of singular values must be less than the smaller matrix dimension for IRLBA iterations");
93 }
94
95 // Falling back to an exact SVD for small matrices or if the requested number is too large
96 // (not enough of a workspace). Hey, I don't make the rules.
97 if ((options.exact_for_small_matrix && smaller < 6) || (options.exact_for_large_number && requested_number * 2 >= smaller)) {
98 exact(matrix, requested_number, outU, outV, outD);
99 return std::make_pair(true, 0);
100 }
101
102 Eigen::Index work = std::min(requested_number + options.extra_work, smaller);
103
104 EigenMatrix_ V(matrix.cols(), work);
105 std::mt19937_64 eng(options.seed);
106 if (options.initial) {
107 auto init = reinterpret_cast<EigenVector_*>(options.initial);
108 if (init->size() != V.rows()) {
109 throw std::runtime_error("initialization vector does not have expected number of rows");
110 }
111 V.col(0) = *init;
112 } else {
113 fill_with_random_normals(V, 0, eng);
114 }
115 V.col(0) /= V.col(0).norm();
116
117 bool converged = false;
118 int iter = 0;
119 Eigen::Index k = 0;
120 JacobiSVD<EigenMatrix_> svd(work, work);
121
122 LanczosWorkspace<EigenVector_, Matrix_> lpwork(matrix);
123
124 EigenMatrix_ W(matrix.rows(), work);
125 EigenMatrix_ Wtmp(matrix.rows(), work);
126 EigenMatrix_ Vtmp(matrix.cols(), work);
127
128 EigenMatrix_ B(work, work);
129 B.setZero(work, work);
130 EigenVector_ res(work);
131 EigenVector_ F(matrix.cols());
132
133 EigenVector_ prevS(work);
134 typename EigenMatrix_::Scalar svtol = options.singular_value_ratio_tolerance;
135 typename EigenMatrix_::Scalar tol = options.convergence_tolerance;
136 typename EigenMatrix_::Scalar svtol_actual = (svtol >= 0 ? svtol : tol);
137
138 for (; iter < options.max_iterations; ++iter) {
139 // Technically, this is only a 'true' Lanczos bidiagonalization
140 // when k = 0. All other times, we're just recycling the machinery,
141 // see the text below Equation 3.11 in Baglama and Reichel.
142 run_lanczos_bidiagonalization(lpwork, W, V, B, eng, k, options);
143
144// if (iter < 2) {
145// std::cout << "B is currently:\n" << B << std::endl;
146// std::cout << "W is currently:\n" << W << std::endl;
147// std::cout << "V is currently:\n" << V << std::endl;
148// }
149
150 svd.compute(B);
151 const auto& BS = svd.singularValues();
152 const auto& BU = svd.matrixU();
153 const auto& BV = svd.matrixV();
154
155 // Checking for convergence.
156 if (B(work - 1, work - 1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
157 converged = true;
158 break;
159 }
160
161 const auto& F = lpwork.F; // i.e., the residuals, see Algorithm 2.1 of Baglama and Reichel.
162 auto R_F = F.norm();
163
164 // Computes the convergence criterion defined in on the LHS of
165 // Equation 2.13 of Baglama and Riechel. (e_m is literally the unit
166 // vector along the m-th dimension, so it ends up just being the
167 // m-th row of the U matrix.) We expose it here as we will be
168 // re-using the same values to update B, see below.
169 res = R_F * BU.row(work - 1);
170
171 Eigen::Index n_converged = 0;
172 if (iter > 0) {
173 auto Smax = *std::max_element(BS.begin(), BS.end());
174 auto threshold = Smax * tol;
175
176 for (Eigen::Index j = 0; j < work; ++j) {
177 if (std::abs(res[j]) <= threshold) {
178 auto ratio = std::abs(BS[j] - prevS[j]) / BS[j];
179 if (ratio <= svtol_actual) {
180 ++n_converged;
181 }
182 }
183 }
184
185 if (n_converged >= requested_number) {
186 converged = true;
187 break;
188 }
189 }
190 prevS = BS;
191
192 // Setting 'k'. This looks kinda weird, but this is deliberate,
193 // see the text below Algorithm 3.1 of Baglama and Reichel.
194 //
195 // - Our n_converged is their k'.
196 // - Our requested_number is their k.
197 // - Our work is their m.
198 // - Our k is their k + k''.
199 if (n_converged + requested_number > k) {
200 k = n_converged + requested_number;
201 }
202 if (k > work - 3) {
203 k = work - 3;
204 }
205 if (k < 1) {
206 k = 1;
207 }
208
209 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
210 V.leftCols(k) = Vtmp.leftCols(k);
211
212 // See Equation 3.2 of Baglama and Reichel, where our 'V' is their
213 // 'P', and our 'F / R_F' is their 'p_{m+1}' (Equation 2.2). 'F'
214 // was orthogonal to the old 'V' and so it should still be
215 // orthogonal to the new left-most columns of 'V'; the input
216 // expectations of the Lanczos bidiagonalization are still met, so
217 // V is ok to use in the next run_lanczos_bidiagonalization().
218 V.col(k) = F / R_F;
219
220 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
221 W.leftCols(k) = Wtmp.leftCols(k);
222
223 B.setZero(work, work);
224 for (Eigen::Index l = 0; l < k; ++l) {
225 B(l, l) = BS[l];
226
227 // This assignment looks weird but is deliberate, see Equation
228 // 3.6 of Baglama and Reichel. Happily, this is the same value
229 // used to determine convergence in 2.13, so we just re-use it.
230 // (See the equation just above Equation 3.5; I think they
231 // misplaced a tilde on the final 'u', given no other 'u' has
232 // a B_m superscript as well as a tilde.)
233 B(l, k) = res[l];
234 }
235 }
236
237 // See Equation 2.11 of Baglama and Reichel for how to get from B's
238 // singular triplets to mat's singular triplets.
239 outD.resize(requested_number);
240 outD = svd.singularValues().head(requested_number);
241
242 outU.resize(matrix.rows(), requested_number);
243 outU.noalias() = W * svd.matrixU().leftCols(requested_number);
244
245 outV.resize(matrix.cols(), requested_number);
246 outV.noalias() = V * svd.matrixV().leftCols(requested_number);
247
248 return std::make_pair(converged, iter + 1);
249}
