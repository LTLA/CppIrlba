40void exact(const Matrix_& matrix, int requested_number, EigenMatrix_& outU, EigenMatrix_& outV, EigenVector_& outD) {
41 JacobiSVD<EigenMatrix_> svd(matrix.rows(), matrix.cols());
42
43 if constexpr(can_svd<EigenMatrix_, Matrix_>::value) {
44 svd.compute(matrix);
45 } else {
46 auto adjusted = wrapped_realize<EigenMatrix_>(matrix);
47 svd.compute(adjusted);

49
50 outD.resize(requested_number);
51 outD = svd.singularValues().head(requested_number);

56 outV.resize(matrix.cols(), requested_number);

93template<class Matrix_, class EigenMatrix_, class EigenVector_>
94std::pair<bool, int> compute(const Matrix_& matrix, Eigen::Index number, EigenMatrix_& outU, EigenMatrix_& outV, EigenVector_& outD, const Options& options) {
95 Eigen::Index smaller = std::min(matrix.rows(), matrix.cols());
96 Eigen::Index requested_number = number;
97 if (requested_number > smaller) {
98 if (options.cap_number) {
99 requested_number = smaller;
100 } else {
101 throw std::runtime_error("requested number of singular values cannot be greater than the smaller matrix dimension");
102 }
103 } else if (requested_number == smaller && !options.exact_for_large_number) {
104 throw std::runtime_error("requested number of singular values must be less than the smaller matrix dimension for IRLBA iterations");
105 }
106
107 // Falling back to an exact SVD for small matrices or if the requested number is too large
108 // (not enough of a workspace). Hey, I don't make the rules.
109 if ((options.exact_for_small_matrix && smaller < 6) || (options.exact_for_large_number && requested_number * 2 >= smaller)) {
110 internal::exact(matrix, requested_number, outU, outV, outD);
111 return std::make_pair(true, 0);
112 }
113
114 Eigen::Index work = std::min(requested_number + options.extra_work, smaller);
115
116 EigenMatrix_ V(matrix.cols(), work);
117 std::mt19937_64 eng(options.seed);
118 if (options.initial) {
119 auto init = reinterpret_cast<EigenVector_*>(options.initial);
120 if (init->size() != V.rows()) {
121 throw std::runtime_error("initialization vector does not have expected number of rows");
122 }
123 V.col(0) = *init;
124 } else {
125 internal::fill_with_random_normals(V, 0, eng);
126 }
127 V.col(0) /= V.col(0).norm();
128
129 bool converged = false;
130 int iter = 0;
131 Eigen::Index k = 0;
132 internal::JacobiSVD<EigenMatrix_> svd(work, work);
133
134 internal::LanczosWorkspace<EigenVector_, Matrix_> lptmp(matrix);
135
136 EigenMatrix_ W(matrix.rows(), work);
137 EigenMatrix_ Wtmp(matrix.rows(), work);
138 EigenMatrix_ Vtmp(matrix.cols(), work);
139
140 EigenMatrix_ B(work, work);
141 B.setZero(work, work);
142 EigenVector_ res(work);
143 EigenVector_ F(matrix.cols());
144
145 EigenVector_ prevS(work);
146 typename EigenMatrix_::Scalar svtol = options.singular_value_ratio_tolerance;
147 typename EigenMatrix_::Scalar tol = options.convergence_tolerance;
148 typename EigenMatrix_::Scalar svtol_actual = (svtol >= 0 ? svtol : tol);
149
150 for (; iter < options.max_iterations; ++iter) {
151 // Technically, this is only a 'true' Lanczos bidiagonalization
152 // when k = 0. All other times, we're just recycling the machinery,
153 // see the text below Equation 3.11 in Baglama and Reichel.
154 internal::run_lanczos_bidiagonalization(matrix, W, V, B, eng, lptmp, k, options);
155
156// if (iter < 2) {
157// std::cout << "B is currently:\n" << B << std::endl;
158// std::cout << "W is currently:\n" << W << std::endl;
159// std::cout << "V is currently:\n" << V << std::endl;
160// }
161
162 svd.compute(B);
163 const auto& BS = svd.singularValues();
164 const auto& BU = svd.matrixU();
165 const auto& BV = svd.matrixV();
166
167 // Checking for convergence.
168 if (B(work - 1, work - 1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
169 converged = true;
170 break;
171 }
172
173 const auto& F = lptmp.F; // i.e., the residuals, see Algorithm 2.1 of Baglama and Reichel.
174 auto R_F = F.norm();
175
176 // Computes the convergence criterion defined in on the LHS of
177 // Equation 2.13 of Baglama and Riechel. (e_m is literally the unit
178 // vector along the m-th dimension, so it ends up just being the
179 // m-th row of the U matrix.) We expose it here as we will be
180 // re-using the same values to update B, see below.
181 res = R_F * BU.row(work - 1);
182
183 Eigen::Index n_converged = 0;
184 if (iter > 0) {
185 auto Smax = *std::max_element(BS.begin(), BS.end());
186 auto threshold = Smax * tol;
187
188 for (Eigen::Index j = 0; j < work; ++j) {
189 if (std::abs(res[j]) <= threshold) {
190 auto ratio = std::abs(BS[j] - prevS[j]) / BS[j];
191 if (ratio <= svtol_actual) {
192 ++n_converged;
193 }
194 }
195 }
196
197 if (n_converged >= requested_number) {
198 converged = true;
199 break;
200 }
201 }
202 prevS = BS;
203
204 // Setting 'k'. This looks kinda weird, but this is deliberate,
205 // see the text below Algorithm 3.1 of Baglama and Reichel.
206 //
207 // - Our n_converged is their k'.
208 // - Our requested_number is their k.
209 // - Our work is their m.
210 // - Our k is their k + k''.
211 if (n_converged + requested_number > k) {
212 k = n_converged + requested_number;
213 }
214 if (k > work - 3) {
215 k = work - 3;
216 }
217 if (k < 1) {
218 k = 1;
219 }
220
221 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
222 V.leftCols(k) = Vtmp.leftCols(k);
223
224 // See Equation 3.2 of Baglama and Reichel, where our 'V' is their
225 // 'P', and our 'F / R_F' is their 'p_{m+1}' (Equation 2.2). 'F'
226 // was orthogonal to the old 'V' and so it should still be
227 // orthogonal to the new left-most columns of 'V'; the input
228 // expectations of the Lanczos bidiagonalization are still met, so
229 // V is ok to use in the next run_lanczos_bidiagonalization().
230 V.col(k) = F / R_F;
231
232 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
233 W.leftCols(k) = Wtmp.leftCols(k);
234
235 B.setZero(work, work);
236 for (Eigen::Index l = 0; l < k; ++l) {
237 B(l, l) = BS[l];
238
239 // This assignment looks weird but is deliberate, see Equation
240 // 3.6 of Baglama and Reichel. Happily, this is the same value
241 // used to determine convergence in 2.13, so we just re-use it.
242 // (See the equation just above Equation 3.5; I think they
243 // misplaced a tilde on the final 'u', given no other 'u' has
244 // a B_m superscript as well as a tilde.)
245 B(l, k) = res[l];
246 }
247 }
248
249 // See Equation 2.11 of Baglama and Reichel for how to get from B's
250 // singular triplets to mat's singular triplets.
251 outD.resize(requested_number);
252 outD = svd.singularValues().head(requested_number);
253
254 outU.resize(matrix.rows(), requested_number);
255 outU.noalias() = W * svd.matrixU().leftCols(requested_number);
256
257 outV.resize(matrix.cols(), requested_number);
258 outV.noalias() = V * svd.matrixV().leftCols(requested_number);
259
260 return std::make_pair(converged, iter + 1);
261}

352template<class Matrix_, class EigenMatrix_, class EigenVector_>
353std::pair<bool, int> compute(const Matrix_& matrix, bool center, bool scale, Eigen::Index number, EigenMatrix_& outU, EigenMatrix_& outV, EigenVector_& outD, const Options& options) {
354 if (!scale && !center) {
355 return compute(matrix, number, outU, outV, outD, options);
356 }
357
358 auto nr = matrix.rows();
359 auto nc = matrix.cols();
360
361 Eigen::VectorXd center0;
362 if (center) {
363 if (nr < 1) {
364 throw std::runtime_error("cannot center with no observations");
365 }
366 center0.resize(nc);
367 }
368
369 Eigen::VectorXd scale0;
370 if (scale) {
371 if (nr < 2) {
372 throw std::runtime_error("cannot scale with fewer than two observations");
373 }
374 scale0.resize(nc);
375 }
376
377 for (Eigen::Index i = 0; i < nc; ++i) {
378 double mean = 0;
379 if (center) {
380 mean = matrix.col(i).sum() / nr;
381 center0[i] = mean;
382 }
383 if (scale) {
384 EigenVector_ current = matrix.col(i); // force it to be a Vector, even if it's a sparse matrix.
385 typename EigenMatrix_::Scalar var = 0;
386 for (auto x : current) {
387 var += (x - mean)*(x - mean);
388 }
389
390 if (var) {
391 scale0[i] = std::sqrt(var/(nr - 1));
392 } else {
393 scale0[i] = 1;
394 }
395 }
396 }
397
398 if (center) {
399 Centered<Matrix_, EigenVector_> centered(matrix, center0);
400 if (scale) {
401 auto centered_scaled = make_Scaled<true>(centered, scale0, true);
402 return compute(centered_scaled, number, outU, outV, outD, options);
403 } else {
404 return compute(centered, number, outU, outV, outD, options);
405 }
406 } else {
407 auto scaled = make_Scaled<true>(matrix, scale0, true);
408 return compute(scaled, number, outU, outV, outD, options);
409 }
410}
